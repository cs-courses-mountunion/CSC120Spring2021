 18     public MUFrame() {
 19         initComponents();
 20         
 21         arr = new Double[7];
 22         
 23         for (int i = 0; i < arr.length; i++) {
 24             arr[i] = (i-3)*(i-3) + i/10.0;
 25             outputArea.append("arr[" + i + "] = " + arr[i] + "\n");
 26         } // end for
 27         
 28         
 29         // find sum
 30         Double sum = 0.0;
 31         for (int i = 0; i < arr.length; i++) {
 32             sum = sum + arr[i];
 33             outputArea.append("\n      (sum so far = " + sum + ")");
 34         } // end for
 35         outputArea.append("\n");
 36         outputArea.append("   The sum of all array elements is " + sum);
 37 
 38         
 39         // find average
 40         // first, find sum, then:
 41         Double avg = sum / arr.length;
 42         outputArea.append("\n");
 43         outputArea.append("   The average of all array elements is " + avg);
 44         
 45         
 46         // find max
 47         Double max = Double.MIN_VALUE;
 48         for (int i = 0; i < arr.length; i++) {
 49             if (arr[i] > max) {
 50                 max = arr[i];
 51             } // end if
 52             outputArea.append("\n       (i = " + i + ", max so far = " + max + ")");
 53         } // end for
 54         outputArea.append("\n");
 55         outputArea.append("   The maximum of all array elements is " + max);
 56         
 57         
 58         // find min
 59         Double min = Double.MAX_VALUE;
 60         for (int i = 0; i < arr.length; i++) {
 61             if (arr[i] < min) {
 62                 min = arr[i];
 63             } // end if
 64             outputArea.append("\n       (i = " + i + ", min so far = " + min + ")");
 65         } // end for
 66         outputArea.append("\n");
 67         outputArea.append("   The minimum of all array elements is " + min);
 68         outputArea.append("\n\n-------------------------\n\n");

 79     public void processSoccerStats() {
 80         
 81         SoccerTeam[] team;
 82         team = new SoccerTeam[7];
 83         team[0] = new SoccerTeam(2019, 8, 10);
 83         team[1] = new SoccerTeam(2018, 12, 7);
 84         team[2] = new SoccerTeam(2017, 12, 6);
 85         team[3] = new SoccerTeam(2016, 11, 6);
 86         team[4] = new SoccerTeam(2015, 8, 9);
 87         team[5] = new SoccerTeam(2014, 4, 14);
 88         team[6] = new SoccerTeam(2013, 5, 13);
 89         
 90         for (int t = 0; t < team.length; t++) {
 91             outputArea.append(team[t].toString() + "\n");
 92         }
 93 
 94         
 95         Integer totWins, minWins, maxLosses;
 96         Double avgWins;
 97         
 98         totWins = 0;
 99         minWins = Integer.MAX_VALUE;
100         maxLosses = Integer.MIN_VALUE;
101         
102         for (int t = 0; t < team.length; t++) {
103             totWins += team[t].getWins();
104             
105             if ( team[t].getWins() < minWins ) {
106                 minWins = team[t].getWins();
107             } // end if
108             
109             if ( team[t].getLosses() > maxLosses ) {
110                 maxLosses = team[t].getLosses();
111             } // end if
112         } // end for
113         
114         avgWins = 1.0 * totWins / team.length;
115         
116         outputArea.append("\n");
117         outputArea.append("  Total Wins in array = " + totWins + "\n");
118         outputArea.append("  Average Wins in array = " + avgWins + "\n");
119         outputArea.append("  Minimum Wins in array = " + minWins + "\n");
120         outputArea.append("  Maximum Losses in array = " + maxLosses + "\n");
121         
122     } // end of processSoccerStats()
