147     private void processDataButtonActionPerformed(java.awt.event.ActionEvent evt) {                                                  
148         
149         String name = nameField.getText();
150         String intValue = integerField.getText();
151         String doubleValue = doubleField.getText();
152         
153         outputArea.setText("Hello there, " + name + "\n");
154         if (! name.equals("Sam Jones")) {
155             outputArea.append("  hey -- you're not Sam Jones...\n");
156         }
157         else {
158             outputArea.append("  glad to see you, Sammy!\n");
159         }
160         
161         Integer x = Integer.parseInt( intValue );
162         Double d  = Double.parseDouble( doubleValue );
163         
164         outputArea.append("\n");
165         outputArea.append("The sum of " + intValue + " and " + doubleValue
166                + " is " + (x + d) + "\n");
167         
168         // add x to the running total
169         total += x;
170         
171     }
